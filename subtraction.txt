.model small
.stack 200h
.data
        prompt1 db 'Enter the first positive integer : $'
        prompt2 db 'Enter the second positive integer : $'
        prompt3 db 'The difference is: $', 0ah, 0dh
.code
.startup
        lea dx, prompt1
        mov ah, 09h
        int 21h

        mov ah, 01h
        int 21h

        mov bl, al
        int 21h

        lea dx, prompt2
        mov ah, 09h
        int 21h

        mov ah, 01h
        int 21h

        mov cl, al
        int 21h

        sub bl, cl

        lea dx, prompt3
        mov ah, 09h
        int 21h

        mov ah, 02h

        add bl, 48
        mov dl, bl
        int 21h

        mov ah, 4ch
        int 21h
    end
